#3.3.6 格式化字符
s = ‘hello python’
x = len(s)
print(‘The lenth of %s is %d’%(s,x)) # 一个%代表一次处理，按照’’外面的括弧内处理，格式化处理就是以某种方式处理元组中的数据
pi = 3.1415926535
print(‘pi = %5.f’%(3,pi)) # 处理方式为对值 pi 进行格式化处理
‘’’
在28line中，’.’之前的数据代表最小宽度，之后的数据代表小数点后面的精度，如果有符号’‘
那么其值需要从元组中第一个数据读出
‘’’
print(‘pi = %.3f’%(10,pi))
print(‘%-10.3f’%pi) # 相比上一行实行了左对齐
print(‘%+f’%pi) # 类型f的默认精度为6
print(‘%010.3f’%pi) # 用0来填补空白，字段宽度为10，精度为3不足补充0
